-- // 14 (Clone + Swap logic) //
local function doCloneSwap()
	if cloning then return end
	cloning = true
	cloneBtn.Text = "Cloning..."
	cloneBtn.BackgroundColor3 = Color3.fromRGB(220, 220, 220)

	pcall(function()
		local char = lp.Character
		if not char then cloning = false; cloneBtn.Text = "Clone + Swap"; return end

		local humanoid = char:FindFirstChildOfClass("Humanoid")
		if not humanoid then cloning = false; cloneBtn.Text = "Clone + Swap"; return end

		-- // Clone the character model //
		local clone = char:Clone()
		clone.Name = lp.Name .. "_Clone"
		clone.Parent = workspace

		-- // Remove scripts from clone //
		for _, v in ipairs(clone:GetDescendants()) do
			if v:IsA("BaseScript") then v:Destroy() end
		end

		-- // Position clone at player location //
		local hrp = char:FindFirstChild("HumanoidRootPart")
		local cloneHRP = clone:FindFirstChild("HumanoidRootPart")
		if hrp and cloneHRP then
			cloneHRP.CFrame = hrp.CFrame * CFrame.new(0, 0, -5)
		end

		-- // Swap: move player to clone position //
		task.wait(0.3)
		if hrp and cloneHRP then
			hrp.CFrame = cloneHRP.CFrame
		end

		-- // Clean up clone after delay //
		task.delay(3, function()
			if clone and clone.Parent then
				clone:Destroy()
			end
		end)
	end)

	task.wait(0.5)
	cloneBtn.Text = "Clone + Swap"
	cloneBtn.BackgroundColor3 = Color3.fromRGB(240, 240, 240)
	cloning = false
end

-- // 15 (Clone button click + hover) //
cloneBtn.MouseButton1Click:Connect(doCloneSwap)

cloneBtn.MouseEnter:Connect(function()
	if not cloning then
		cloneBtn.BackgroundColor3 = Color3.fromRGB(220, 220, 220)
	end
end)
cloneBtn.MouseLeave:Connect(function()
	if not cloning then
		cloneBtn.BackgroundColor3 = Color3.fromRGB(240, 240, 240)
	end
end)

-- // 16 (Keybind button — click to rebind) //
kbBtn.MouseButton1Click:Connect(function()
	waitingForKey = true
	kbBtn.Text = "[...]"
	kbBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
end)

-- // 17 (Input listener — keybind + trigger) //
UserInputService.InputBegan:Connect(function(input, gpe)
	if gpe then return end
	if UserInputService:GetFocusedTextBox() then return end

	if waitingForKey then
		if input.UserInputType == Enum.UserInputType.Keyboard then
			keybind = input.KeyCode
			kbBtn.Text = "[" .. keybind.Name .. "]"
			kbBtn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
			waitingForKey = false
		end
		return
	end

	if input.KeyCode == keybind then
		doCloneSwap()
	end
end)
